1    # Function calculates the minimum difference/cost between 2 strings X and Y (X -> Y)
2    def EditDistance(X, m, Y, n):
3        distance = [[0 for _ in range(n+1)] for _ in range(m+1)]
4    
5        for i in range(m + 1):
6            distance[i][0] = i
7    
8        for i in range(n + 1):
9            distance[0][i] = i
10   
11       for i in range(1, m + 1):
12           for j in range(1, n + 1):
13               if X[i - 1] == Y[j - 1]:
14                   distance[i][j] = distance[i - 1][j - 1]
15               else:
16                   distance[i][j] = 1 + min(distance[i - 1][j],    # Delete letter from X
17                                            distance[i][j - 1],    # Insert letter from Y to X
18                                            distance[i - 1][j - 1])     # Replace letter in X with Y
19   
20       return distance[m][n]

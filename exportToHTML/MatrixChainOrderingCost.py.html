1    # Function to find the most efficient way to multiply given sequence of matrices
2    import sys
3    def MatrixChainOrderingCost(dimensions):
4        n = len(dimensions)
5    
6        dp = [[0 for _ in range(n)] for _ in range(n)]
7        # dp[i,j] = minimum number of scalar multiplications (i.e., cost)
8        # needed to compute the matrix M[i]M[i+1]...M[j] = M[i..j]
9        # The cost is zero when multiplying one matrix <====
10   
11       # for each kind of possible split, we calculate cost and find the minimum
12       for length in range(2, n):
13           for start in range(1, n - length + 1):
14               end = start + length - 1
15               dp[start][end] = sys.maxsize
16               for k in range(start, end):
17                   # split and find for M[start..k] and M[k..end] and add them with the cost
18                   cost = dp[start][k] + dp[k + 1][end] + \
19                          dimensions[start - 1] * dimensions[k] * dimensions[end]
20                   if cost < dp[start][end]:
21                       dp[start][end] = cost
22       return dp[1][n - 1]
23

26   dimensions = [10, 30, 5, 60]

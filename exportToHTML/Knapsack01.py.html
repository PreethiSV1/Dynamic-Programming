1    # Function to calculate total value that can be gained from the knapsack of given capacity
2    def Knapsack(value, weight, n, capacity):
3        for i in range(n + 1):
4            for w in range(capacity + 1):
5                if i == 0 or w == 0:
6                    dp[i][w] = 0
7                elif weights[i - 1] <= capacity:
8                    # if that weight can be put inside the knapsack
9                    # maximum among the profits of including and excluding the current weight
10                   dp[i][w] = max(value[i - 1] + dp[i - 1][capacity - weight[i - 1]],  dp[i - 1][capacity])
11               else:
12                   dp[i][w] = dp[i - 1][w]
13       return dp[n][capacity]

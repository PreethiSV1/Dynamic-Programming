1    # Function to find maximum sum of increasing subsequence in an array
2    # eg: [8, 4, 12, 2, 10, 6, 14, 1, 9, 5, 13, 3, 11] => 34 ==> [8, 12, 14]
3    def MaximumSumIncreasingSubsequence(X, m):
4        sumDp = [0 for _ in range(m)]
5    
6        # to store (maximum sum increasing) subsequence till each i-th index
7        MSIS = [[] for _ in range(m)]
8    
9        # sum will be the number itself at index 0
10       sumDp[0] = X[0]
11       MSIS[0].append(X[0])
12   
13       for i in range(1, m):
14           for j in range(i):
15               if X[i] > X[j] and sumDp[i] < sumDp[j]:
16                   MSIS[i] = MSIS[j].copy()
17                   sumDp[i] = sumDp[j]
18           # include i-th element in the sum
19           sumDp[i] += X[i]
20           MSIS[i].append(X[i])
21   
22       maxSum = 0
23       for i in range(1, m):
24           if sumDp[i] > sumDp[maxSum]:
25               maxSum = i
26   
27       return sumDp[maxSum], MSIS[maxSum]
28

30   X = [8, 4, 12, 2, 10, 6, 14, 1, 9, 5, 13, 3, 11]
31   m = len(X)
32   maxSum, maxSequence = MaximumSumIncreasingSubsequence(X, m)
33   print(maxSum)
34   print(maxSequence)

1    # Function to calculate minimum cost to reach a cell at position (m,n) in any matrix
2    def MinimumCostPath(cost, m, n):
3        dp = [[0 for _ in range(n)] for _ in range(m)]
4    
5        dp[0][0] = cost[0][0]
6        for i in range(1, m):
7            dp[i][0] = dp[i - 1][0] + cost[i][0]
8        for j in range(1, n):
9            dp[0][j] = dp[0][j - 1] + cost[0][j]
10       for i in range(1, m):
11           for j in range(1, n):
12               # only right (dp[i - 1][j]) and down (dp[i][j - 1]) moves are possible
13               # to include moving diagonally, include dp[i - 1][j - 1]
14               dp[i][j] = min(dp[i - 1][j], dp[i][j - 1]) + cost[i][j]
15       return dp[m - 1][n - 1]

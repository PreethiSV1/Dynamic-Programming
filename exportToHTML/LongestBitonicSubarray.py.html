1    # Function to find longest bitonic subarray (continuous numbers --> subarray)
2    # bitonic ---> Increases and then decreases eg: 1 4 7 10 6 3 (1 < 4 < 7 < 10 > 6 > 3)
3    def LongestBitonicSubarray(X, m):
4        LIS = [0 for _ in range(m)]
5        LDS = [0 for _ in range(m)]
6    
7        # from first number
8        LIS[0] = 1
9        for i in range(1, m):
10           LIS[i] = 1
11           if X[i - 1] < X[i]:
12               # only if prev number is smaller than current
13               LIS[i] = LIS[i - 1] + 1
14   
15       # from last number
16       LDS[m - 1] = 1
17       for i in reversed(range(m - 1)):
18           LDS[i] = 1
19           if X[i] > X[i + 1]:
20               # only if next number is smaller than current
21               LDS[i] = LDS[i + 1] + 1
22   
23       longestBSLength = 1
24       beg, end = 0, 0
25       for i in range(0, m):
26           if longestBSLength < LDS[i] + LIS[i] - 1:
27               longestBSLength = LDS[i] + LIS[i] - 1
28               beg = i - LIS[i] + 1  # current index - length of LIS at i
29               end = i + LDS[i] - 1  # current index + length of LDS at i
30   
31       return longestBSLength, X[beg: end + 1]
32   
33   
34   X = [3, 5, 8, 4, 5, 9, 10, 8, 5, 3, 4]
35   m = len(X)
36   length, subarray = LongestBitonicSubarray(X, m)
37   print(length)

1    # Function to simply find the length of LIS without using LCS
2    def LongestIncreasingSubsequenceLength(X, m):
3        dp = [0 for i in range(m)]
4    
5        # starting with index 0 => LIS length = 1
6        dp[0] = 1
7    
8        for i in range(1, m):
9            dp[i] = 1
10           for j in range(i):
11               if X[i] > X[j] and dp[i] < dp[j]:
12                   # If current number is greater than the previous,
13                   # copy the previous number's LIS value
14                   dp[i] = dp[j]
15           # include i-th value in the sequence length
16           dp[i] += 1
17       return max(dp)

1    # Function to find longest bitonic subsequence length (need not be continuous)
2    # bitonic ---> Increases and then decreases eg: 1 4 7 10 6 3 (1 < 4 < 7 < 10 > 6 > 3)
3    def LongestBitonicSubsequenceLength(X, m):
4        LIS = [0 for _ in range(m)]
5        LDS = [0 for _ in range(m)]
6    
7        # from first number
8        LIS[0] = 1
9        for i in range(1, m):
10           for j in range(i):
11               if X[i] > X[j] and LIS[i] < LIS[j]:
12                   LIS[i] = LIS[j]
13           LIS[i] += 1
14   
15       # from last number
16       LDS[m - 1] = 1
17       for i in reversed(range(m - 1)):
18           for j in reversed(range(i - 1, m)):
19               if X[i] > X[j] and LDS[i] < LDS[j]:
20                   LDS[i] = LDS[j]
21           LDS[i] += 1
22   
23       longestBSLength = LIS[0] + LDS[0] - 1
24       for i in range(1, m):
25           longestBSLength = max(longestBSLength, LIS[i] + LDS[i] - 1)
26   
27       return longestBSLength

30   def LongestBitonicSubsequence(X, m):
31       LIS = [[] for _ in range(m)]
32       LDS = [[] for _ in range(m)]
33   
34       # from first number
35       LIS[0].append(X[0])
36       for i in range(1, m):
37           for j in range(i):
38               if X[i] > X[j] and len(LIS[i]) < len(LIS[j]):
39                   LIS[i] = LIS[j].copy()
40           LIS[i].append(X[i])
41   
42       # from last number
43       LDS[m - 1].append(X[m - 1])
44       for i in reversed(range(m - 1)):
45           for j in reversed(range(i - 1, m)):
46               if X[i] > X[j] and len(LDS[i]) < len(LDS[j]):
47                   LDS[i] = LDS[j].copy()
48           LDS[i].insert(0, X[i])
49   
50       peak = 0
51       for i in range(1, m):
52           if (len(LIS[i]) + len(LDS[i])) > (len(LIS[peak]) + len(LDS[peak])):
53               peak = i    # Maximum increasing followed by decreasing sequence's position
54   
55       return LIS[peak][:-1] + LDS[peak]    # since peak index's value will be present in both

1    # Modification of Knapsack Problem <==== Important
2    # Function to check if a subset with given sum is possible
3    def SubsetSum(arr, n, sum):
4        dp = [[False for _ in range(sum + 1)] for _ in range(n + 1)]
5    
6        # if 0 items in the list and sum is non-zero
7        for i in range(sum + 1):
8            dp[0][i] = False
9    
10       # if sum is zero
11       for i in range(n + 1):
12           dp[i][0] = True
13   
14       for i in range(1, n + 1):
15           for j in range(1, sum + 1):
16               if arr[i - 1] > j:  # don't include ith element if j - arr[i-1] is negative
17                   dp[i][j] = dp[i - 1][j]
18               else:
19                   # can a subset be formed
20                   # either including OR excluding the current number
21                   dp[i][j] = dp[i - 1][j - arr[i - 1]] or dp[i - 1][j]
22   
23       return dp[n][sum]

28   requiredSum = 18

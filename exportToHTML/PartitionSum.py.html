2    # Partition problem - Return true if given array arr[0..n-1] can
3    # be divided into two subsets with equal sum
4    def PartitionSum(arr):
5        total = sum(arr)
6        return (not (total % 2)) and SubsetSum(arr, len(arr), total // 2)

9    # Function to check if a subset with given sum is possible
10   def SubsetSum(arr, n, sum):
11       dp = [[False for _ in range(sum + 1)] for _ in range(n + 1)]
12   
13       # if number of items(n)is zero and sum is non zero
14       for i in range(sum + 1):
15           dp[0][i] = False
16   
17       # if sum is zero and n is non zero
18       for i in range(n + 1):
19           dp[i][0] = True
20   
21       for i in range(1, n + 1):
22           for j in range(1, sum + 1):
23               if arr[i - 1] > j:
24                   dp[i][j] = dp[i - 1][j]
25               else:
26                   # either including (decrement the sum) OR
27                   # excluding the element can give the required subset
28                   dp[i][j] = dp[i - 1][j - arr[i - 1]] or dp[i - 1][j]
29   
30       return dp[n][sum]
